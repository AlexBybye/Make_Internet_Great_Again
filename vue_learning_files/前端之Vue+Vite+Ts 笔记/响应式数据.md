# 既然更有效率，那么我们遇到父组件是reactive，子组件如何调用呢？
## toRefs || toRef
```vue
<template>
  <br>
  <h2>Game List</h2>
  <ul>
    <li v-for="g in games" :key="g.id">{{ g.name }}</li>
  </ul>
  <button @click="changeSecondgame">修改第二个游戏名字</button>
  <div class="pet">
    <h2>宠物信息：</h2>
    <h3>品种：{{ message.type }}</h3>
    <h3>售价：{{ message.price }}</h3>
    <button @click="changeprice">修改价格</button>
    <button @click="changeall">修改此宠物品种及其售价</button>
  </div>
</template>

<script lang="ts" setup name="mixture">
import { reactive, toRef, toRefs } from 'vue'
// 使用 reactive 定义父组件的数据
const state = reactive({
  message: { type: "中华田园犬", price: 1 },
  games: [
    { id: "auisfhnch", name: 'gta5' },
    { id: "bhaikcsjk", name: 'gta4' },
    { id: "chjanjcv", name: 'gta3' }
  ]
})
// 方法函数
function changeprice() {
  state.message.price = Math.floor(Math.random() * 1000);
}
function changeall() {
  state.message = { type: "东北宇将军", price: 100 };
}
function changeSecondgame() {
  state.games[1].name = '看门狗2';
}

// 使用 toRef 或 toRefs 将数据传递到子组件
// 如果需要传递单个属性，使用 toRef
const messageRef = toRef(state, 'message');

// 如果需要传递多个属性，使用 toRefs
const { games } = toRefs(state);
</script>

<style>
.li {
  font-size: 20px;
}
</style>
```
## 子组件：
```vue
<script lang="ts" setup>
import { defineProps } from 'vue'

// 子组件接收父组件传递的 props
const props = defineProps({
  message: Object,
  games: Array
})

// 在子组件中可以直接使用 props.message 和 props.games
</script>

<template>
  <div class="child-component">
    <h2>子组件 - 宠物信息：</h2>
    <h3>品种：{{ props.message.type }}</h3>
    <h3>售价：{{ props.message.price }}</h3>
    <h2>子组件 - 游戏列表：</h2>
    <ul>
      <li v-for="g in props.games" :key="g.id">{{ g.name }}</li>
    </ul>
  </div>
</template>

<style>
.child-component {
  border: 1px solid #ccc;
  padding: 10px;
  margin-top: 20px;
}
</style>
```
## 这里有个好习惯：vue3习惯性把所有响应式数据都放到state中集中管理，并用reactive来初始化state使性能高效

### reactive自带拆包属性，所以只要外边有reactive，里面的ref搞出来不用 .value
